🔒 TLSNotary Browser Extension Test

Verifier Server: ws://localhost:7048

Setup

  1. Install TLSNotary Extension from Chrome Web Store
  2. Click the extension icon → open DevConsole
  3. In DevConsole, paste the plugin code below and click Run

Plugin Code (copy this to DevConsole)

// CoinGecko BTC Price — Anchr TLSNotary Test
const VERIFIER_URL = 'ws://localhost:7048';
const PROXY_URL = 'ws://localhost:7048/proxy?token=api.coingecko.com';

export default {
  config: {
    name: 'Anchr: BTC Price Verification',
    description: 'Prove Bitcoin USD price from CoinGecko API via Anchr Verifier',
  },
  main: async () => {
    const proof = await prove(
      {
        url: 'https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd',
        method: 'GET',
        headers: { 'Accept': 'application/json', 'Accept-Encoding': 'identity', 'Connection': 'close' },
      },
      {
        verifierUrl: VERIFIER_URL,
        proxyUrl: PROXY_URL,
        maxRecvData: 16384,
        maxSentData: 4096,
        handlers: [
          { type: 'SENT', part: 'START_LINE', action: 'REVEAL' },
          { type: 'RECV', part: 'STATUS_CODE', action: 'REVEAL' },
          { type: 'RECV', part: 'BODY', action: 'REVEAL' },
        ],
      }
    );
    console.log('Proof:', JSON.stringify(proof.results, null, 2));
    return proof;
  },
};
